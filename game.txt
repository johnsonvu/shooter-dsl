make game TouHouV2
{
    width = 1024
    height = 768
}
{
	//create timer variable
	new timer
	timer = 0
	
    // create player
	make player Hero
	
	// create boss
	make enemy MasterMind
	
	//make some other counters
	new myCounter
	myCounter = 0
	new yourCounter
	yourCounter = 0
}

// modify objects
set Hero{
    damage = 30
    health = 10000
}

set MasterMind{
    damage = 1
    health = 10000
    behave = evil
}

set circleBuddy{
    damage = 100
    health = 100
    behave = circleBuddyBehaviour    //Run after every tick in the game
}

set powerUp{
    health = 1000
}

define circleBuddyBehaviour() {
    if(myCounter>0){
        if(myCounter<100){
            move down
            shoot down
        }
    }
    if(myCounter>100 and myCounter<200){ //use "and" instead of nested if statements
		move right
		shoot right
    }
    if(myCounter>200 and myCounter<300){
        move up
        shoot up
    }
    if(myCounter>300 and myCounter<400){
        move left
        shoot left
    }
    if(myCounter>400){
        myCounter = 0
    }

    myCounter = myCounter + 1
}

define evil(){
	if (timer = 0)
	{
		make 10 enemy firstwave
	}
	if (timer = 500)
	{
		make 10 enemy secondwave
	}
	if (timer = 1000)
	{
		make 10 enemy thirdwave
	}
	if (timer = 1500)
	{
		make 10 enemy circleBuddy
	}
	if (timer > 2000)
	{
		move 4 right
		
		if (yourCounter = 41){
			yourCounter = 0
		}
		
		if (yourCounter < 20){
			move 4 down
		}
		if (yourCounter > 19 and yourCounter < 40){
			move 4 up
		}
		
		yourCounter = yourCounter+1
		
		shoot up
		shoot down
		shoot left
		shoot right
		
		damage = call doubleDamage(damage)

		if (damage > 1000){
		    damage = 1000
		}
		
	}
	
	//if they can't kill the boss fast enough, MOAR ENEMIES =)
	if (timer = 4000)
	{
		timer = 0
	}
	
	timer = timer + 1

}

// function def
define doubleDamage(dmg){
    return 2 * dmg
}
